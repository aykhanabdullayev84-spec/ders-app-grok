<script>
const supabase = Supabase.createClient('https://YOUR-PROJECT.supabase.co', 'eyJ...'); // ← buranı dəyiş

supabase.channel('live-admin')
  .on('postgres_changes', { event: '*', schema: 'public', table: 'profiles' }, loadLiveData)
  .subscribe();

async function loadLiveData() {
  const { data } = await supabase.from('profiles').select('name, xp, city').order('xp', { ascending: false }).limit(10);
  document.getElementById('liveCount').textContent = data.length * 5 + 30 + ' nəfər indi öyrənir';
  document.getElementById('leaderboard').innerHTML = data.map(u => `<tr><td>${u.name}</td><td class="text-amber-400">${u.xp} XP</td></tr>`).join('');
}
loadLiveData();
</script>
