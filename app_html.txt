<script>
// ================== SUPABASE CONFIG (BURANI DƏYİŞ) ==================
const SUPABASE_URL = 'https://YOUR-PROJECT.supabase.co';
const SUPABASE_ANON = 'eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCJ9...'; // ← buranı dəyiş

const supabase = Supabase.createClient(SUPABASE_URL, SUPABASE_ANON);

let user = null;
let carpetData = [];

async function initRealtime() {
  const { data: { session } } = await supabase.auth.getSession();
  if (session) user = session.user;

  supabase.channel('carpet-sync')
    .on('postgres_changes', { event: 'UPDATE', schema: 'public', table: 'profiles', filter: `id=eq.${user?.id}` }, (payload) => {
      carpetData = payload.new.carpet_data || [];
      renderCarpet();
      showToast('Xalçan yeniləndi! 🧶');
    })
    .subscribe();

  supabase.channel('broadcast')
    .on('postgres_changes', { event: 'INSERT', schema: 'public', table: 'broadcasts' }, (payload) => {
      showToast(payload.new.message);
      if (payload.new.event === 'lesson_complete') confetti({particleCount:150});
    })
    .subscribe();
}

function showToast(msg) {
  const t = document.getElementById('toast');
  t.textContent = msg;
  t.classList.remove('hidden');
  setTimeout(() => t.classList.add('hidden'), 4000);
}

async function completeLesson() {
  carpetData.push(Date.now());
  await supabase.from('profiles').update({ carpet_data: carpetData }).eq('id', user.id);
  await supabase.from('broadcasts').insert({ event: 'lesson_complete', message: 'Kimisə dərs bitirdi! 🔥' });
  renderCarpet();
  confetti({particleCount:300});
}

function renderCarpet() {
  const grid = document.getElementById('carpet');
  grid.innerHTML = '';
  for (let i = 0; i < 49; i++) {
    const knot = document.createElement('div');
    knot.className = `carpet-knot flex items-center justify-center text-3xl shadow-inner ${(i < carpetData.length) ? 'bg-red-600' : 'bg-zinc-700'}`;
    knot.textContent = i < carpetData.length ? '🧶' : '⬜';
    grid.appendChild(knot);
  }
}

async function startLesson() {
  completeLesson(); // demo
}

initRealtime();
</script>
